26class HISTORY(): 

27 """Implements functions related to managing the history of the conversations.""" 

28 def __init__(self): 

29 """ 

30 Init function for HISTORY class. 

31 """ 

32 # No. of times to retry writing into the file 

33 self.__retries = 5 

34 # Path to store the conversational history 

35 data_folder = os.getcwd() + "/history/" 

36 if not os.path.isdir(data_folder): 

37 os.makedirs(data_folder) 

38 

39 # File name to store all the conversations 

40 self.__file_path = data_folder + "conversations_db.tsv" 

41 # Update the file with the headers, if the file does not already exist. 

42 if not os.path.isfile(self.__file_path): 

43 with codecs.open(self.__file_path, 'w', "utf-8") as fp: 

44 fp.write('time\tuser_input\tllm_response\ttime_to_response\tBLEU_score\n') 

45 

46 def write(self, ts, user_input, llm_response, time_to_response, BLEU_score): 

47 """ 

48 Write the data into the history file. 

49 """ 

50 data = str(ts) + '\t' + user_input + "\t" + llm_response.replace("\n", ' ') + '\t' + \ 

51 str(time_to_response) + '\t' + str(BLEU_score) + '\n' 

52 

53 # If there are multiple instances of this class, all trying to write into the file, 

54 # We want to handle them without throwing an exception, 

55 retries = self.__retries 

56 while retries > 0: 

57 try: 

58 with codecs.open(self.__file_path, 'a', "utf-8") as fp: 

59 fp.write(data) 

60 break 

61 except Exception as err: 

62 time.sleep(1) 

63 retries -= 1

66class LLM: 

67 """Local LLM that is customized to a specific domain/task.""" 

68 def __init__(self, base_instruction="", template_option=1): 

69 """ 

70 Init function for LLM class. 

71 

72 @param: base_instruction 

73 Custom base instruction to be used for the LLM. 

74 When specified, the available prompt templates are ignored. 

75 @param: option 

76 Specify which prompt template to be used as the base instruction for the LLM. 

77 """ 

78 self.doc_obj = DOCUMENTS() 

79 self.his_obj = HISTORY() 

80 self.vec_obj = VECTORDB(main_doc_dir=self.doc_obj.main_doc_dir) 

81 

82 # Path to store the evaluation metrics 

83 evaluation_metrics = os.getcwd() + "/metrics/" 

84 # Input data to calculate evaluation metrics 

85 self.metrics_input_file = evaluation_metrics + "/input.tsv" 

86 # File name to store the BLEU score 

87 self.bleu_score_file = evaluation_metrics + "/BLEU_score.tsv" 

88 # Update the file with the headers, if the file does not already exist. 

89 if not os.path.isfile(self.bleu_score_file): 

90 with codecs.open(self.bleu_score_file, 'w', "utf-8") as fp: 

91 fp.write('user_input\tllm_response\treference_response\tBLEU_score\n') 

92 

93 # Read the Ollama embeddings 

94 embeddings = OllamaEmbeddings(model=self.vec_obj.llm, show_progress=False) 

95 db = Chroma(persist_directory=self.vec_obj.db_dir, 

96 embedding_function=embeddings) 

97 

98 # Create retriever 

99 retriever = db.as_retriever( 

100 search_type="similarity", 

101 search_kwargs= {"k": 5} 

102 ) 

103 

104 # Create local LLM 

105 llm = ChatOllama(model=self.vec_obj.llm, 

106 keep_alive="3h", 

107 max_tokens=2048, 

108 temperature=0) 

109 

110 if base_instruction != "": 

111 template = base_instruction 

112 else: 

113 template = self.__get_prompt_template(template_option) 

114 prompt = ChatPromptTemplate.from_template(template) 

115 

116 # Create the RAG chain with prompt printing and streaming output 

117 self.rag_chain = ( 

118 {"context": retriever, "prompt": RunnablePassthrough()} 

119 | prompt 

120 | llm 

121 )

172 def __calculate_bleu_score(self, reference, candidate): 

173 """ 

174 Calculate the BLEU score for a candidate sentence against a reference sentence. 

175  

176 @param: reference 

177 The reference sentence (a list of words). 

178 @param: candidate 

179 The candidate sentence generated by the model (a list of words). 

180  

181 @return: BLEU score 

182 """ 

183 # Tokenize the reference and candidate if they are not already tokenized 

184 if isinstance(reference, str): 

185 reference = reference.split() 

186 if isinstance(candidate, str): 

187 candidate = candidate.split() 

188 

189 # Calculate BLEU score 

190 smoothing_function = SmoothingFunction().method1 

191 bleu_score = sentence_bleu([reference], candidate, smoothing_function=smoothing_function) 

192 

193 return bleu_score

195 def chat(self, prompt, reference_response=None, store_history=True): 

196 """ 

197 Chat with the LLM. 

198 

199 @param: prompt 

200 Prompt to be send to the LLM for getting a response 

201 

202 @param: reference_response 

203 The reference response for evaluating BLEU score 

204 

205 @param: store_history 

206 Set this to False if you do not want to store the conversational history 

207 

208 @return: return the response from the LLM for the specified prompt. 

209 """ 

210 #print("User input : {}".format(prompt)) 

211 

212 st = time.time() 

213 answer = "" 

214 for chunk in self.rag_chain.stream(prompt): 

215 answer += chunk.content 

216 et = time.time() - st 

217 

218 #print("Cllama ({time}sec) : {ans}".format(time=round(et, 2), ans=answer)) 

219 

220 # if the reference response is given, then calculate BLEU score also 

221 BLEU_score = '' 

222 if reference_response: 

223 BLEU_score = self.__calculate_bleu_score(reference=reference_response, 

224 candidate=answer) 

225 if store_history: 

226 self.his_obj.write(ts=datetime.now(), 

227 user_input=prompt, 

228 llm_response=answer, 

229 time_to_response=et, 

230 BLEU_score=BLEU_score) 

231 

232 return (answer, BLEU_score)

235 """

240 # The input prompt and the references are going to be the documents fed to the LLM

244 with codecs.open(self.bleu_score_file, 'a', "utf-8") as fp:
